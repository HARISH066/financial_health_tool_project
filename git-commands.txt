# Git Commands to Push Your Project

# Step 1: Initialize Git
git init

# Step 2: Add all files
git add .

# Step 3: Commit
git commit -m "Initial commit: Financial Health Assessment Tool for SMEs"

# Step 4: Create GitHub repository
# Go to https://github.com/new
# Create a new repository (e.g., financial-health-assessment)
# Copy the repository URL

# Step 5: Add remote (replace YOUR_USERNAME and YOUR_REPO)
git remote add origin https://github.com/YOUR_USERNAME/YOUR_REPO.git

# Step 6: Push to GitHub
git branch -M main
git push -u origin main

# Done! Your code is now on GitHub.
# Copy the repository URL and add it to README.md
